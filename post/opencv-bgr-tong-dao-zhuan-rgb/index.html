OpenCV BGR通道转RGB

2020-04-23 1 min read # opencv # python 

import cv2

def BGR2RGB(img):
    '''
        BGR通道转换为RGB通道
    :param img:要转换的图片
    :return:转换后的图片
    '''
    b = img[:,:,0].copy()
    # 学习时候有这一句，也是让我迷糊的
    # BGR转RGB，只要将R与B通道转换一下，为什么要写个G，不多于吗？
    # 我找了几张照片实验了一下，有没有这个是一样的。不知道是不是有些图片比较特殊。
    # g = img[:,:,1].copy()
    r = img[:,:,2].copy()

    img[:,:,0] = r
    # img[:,:,1] = g
    img[:,:,2] = b
    return img

# 图片路径
imgpath = 'img\\image1.jpg'

# 读取图片
img = cv2.imread(imgpath)
# 显示原本图片
cv2.imshow('old',img)

# 转换图片
img = BGR2RGB(img)
# 显示转换后的图片
cv2.imshow('new',img)

# 等待
cv2.waitKey()
cv2.destroyAllWindows()
