五星红旗迎风飘扬~ - EasyX库绘制五星红旗

 * 作者： ruxia-TJY

 * 时2019年12月24日18点52分，最后修改代码

#include<graphics.h>
#include<math.h>

#define PI 3.1415926535

bool HasStandardLine = false;	// 是否显示标准线

void DrawingFivePointedStar(POINT centlocation, int radius, double RotationAngle);			// 绘制五角星	
void DrawingFlag();						// 绘制红旗
void standardline();					// 绘制标准线

int main()
{
	MOUSEMSG msg;

	initgraph(600, 400);

	// 设置背景色
	setbkcolor(RED);
	cleardevice();
	DrawingFlag();
	// 绘制红旗

	while(1)
	{
		msg = GetMouseMsg();
		switch (msg.uMsg) {
		case WM_LBUTTONDOWN:
			// 按下鼠标左键
			// 绘制红旗
			DrawingFlag();
			
			// 绘制标准线
			if (!HasStandardLine) {
				standardline();
			}
			HasStandardLine = !HasStandardLine;
			
			break;
		case WM_RBUTTONDOWN:
			// 右键退出
			return 0;
		}
	}	
}

// 绘制红旗
void DrawingFlag()
{
	// 绘制背景
	setfillcolor(RED);
	solidrectangle(0, 0, 600, 400);
	POINT p;

	// 绘制五角星
	p.x = 100;
	p.y = 100;
	DrawingFivePointedStar(p, 60, PI / 2);

	p.x = 200;
	p.y = 40;
	DrawingFivePointedStar(p, 20, PI * 211 / 180);

	p.x = 240;
	p.y = 80;
	DrawingFivePointedStar(p, 20, PI * 188 / 180);

	p.x = 240;
	p.y = 140;
	DrawingFivePointedStar(p, 20, PI * 164 / 180);

	p.x = 200;
	p.y = 180;
	DrawingFivePointedStar(p, 20, PI * 141 / 180);
}

// 绘制五角星
void DrawingFivePointedStar(POINT centlocation, int radius, double RotationAngle)
{
	POINT pts[5];

	for (int i = 0; i < 5; i++)
	{
		pts[i].x = int(centlocation.x + cos(RotationAngle) * radius);
		pts[i].y = int(centlocation.y - sin(RotationAngle) * radius);
		RotationAngle += PI * 4 / 5;
	}

	setpolyfillmode(WINDING);
	setfillcolor(YELLOW);
	solidpolygon(pts, 5);
}

// 绘制标准线
void standardline()
{
	int i;						// 随机变量
	setlinecolor(BLUE);
	line(0, 200, 600, 200);
	line(300, 0, 300, 400);

	for (i = 1; i <= 10; i++) {
		line(0, 20 * i, 300, 20 * i);
	}
	for (i = 1; i <= 15; i++) {
		line(20 * i, 0, 20 * i, 200);
	}

	line(100, 100, 200, 40);
	line(100, 100, 240, 80);
	line(100, 100, 240, 140);
	line(100, 100, 200, 180);

	circle(100, 100, 60);
	circle(200, 40, 20);
	circle(240, 80, 20);
	circle(240, 140, 20);
	circle(200, 180, 20);
}
